#OSC
---- OSC Syntax
OSC-packet  := OSC-content OSC-size
               ; order and structure of contents and size are
               ; depends on data transfer protocol like UDP/TCP
OSC-size    := [bytes size of OSC-content]
OSC-content := OSC-message | OSC-bundle

OSC-message := OSC-address-pattern OSC-type-tag-string *OSC-arguments.
OSC-address-pattern := '/' *. ; OSC-string
OSC-type-tag-string := ',' *OSC-type-tag
               ;OSC-string, represents OSC-arguments.
OSC-type-tag:= 'i' | 'f' | 's' | 'b' | OSC-type-tag-nonstandard
               ; int32, float32, OSC-string, OSC-blob
OSC-type-tag-nonstandard := [htdScrmTFNI\[\]]
               ; http://opensoundcontrol.org/spec-1_0
OSC-arguments := int32 | float32 | OSC-string | OSC-blob | [a binary data represented by OSC-type-tag-nonstandard]

OSC-bundle  := "#bundle" OSC-timetag *OSC-bundle-element
OSC-bundle-element := [int32 size of content] OSC-content

Atomic-data := int32 | OSC-timetag | float32 | OSC-String | OSC-blob
int32       := [32bit big-endian signed integer]
OSC-timetag := [64bit big-endian fixed number]
float32     := [32bit big-endian IEEE754 float]
OSC-string  := [null terminated ASCII string] [null padding to 4 byte segments]
OSC-blob    := int32[size] [binary data] [null padding to 4 byte segments]

---- OSC Semantics OSC-address
ascii-printable := [ !"#$%&'()*+,-./0123456789:;<=>?@ABCDEFGHIJKLMNOPQRSTUVWXYZ[\]^_`abcdefghijklmnopqrstuvwxyz{|}~]

OSC-address   := *('/' OSC-container) '/' OSC-method
OSC-method    := OSC-address-symbol
OSC-container := OSC-address-symbol
OSC-address-symbol := +OSC-address-letter
OSC-address-letter := [!"$%&'()+-.0123456789:;<=>@ABCDEFGHIJKLMNOPQRSTUVWXYZ\^_`abcdefghijklmnopqrstuvwxyz|~]

OSC-address-pattern := *('/' OSC-address-pattern-part)
OSC-address-pattern-part := *OSC-address-pattern-part-token
OSC-address-pattern-part-token := OSC-address-letter | 
                                  '?' ;任意の一文字にマッチ
                                  '*' ;任意の文字列にマッチ
                                  '[' ?'!' +(OSC-address-letter|OSC-address-letter-range) ']' ;文字グループにマッチ
                                  '{' OSC-address-pattern-part *(',' OSC-address-pattern-part) '}' ;カンマ区切りのどれかにマッチ
OSC-address-letter-range       := OSC-address-letter '-' OSC-address-letter

OSC-address-pattern がマッチする全ての OSC-address を持つ OSCメソッド がコールされる。

---- OSC Semantics OSC-timetag
OSC-timetag := immidiate | seconds pseconds
immidiate := 0x0000000000000001
seconds   := [1900-01-01 00:00:00 からの経過秒数]
pseconds  := [p; 0 <= p < 5000000000; means p * 200 pico seconds]
